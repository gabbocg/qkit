% qkit-paper LaTeX preamble for the internet appendix.
% Loaded via the YAML include-in-header field in internet-appendix.qmd.

\usepackage{amsthm}
\usepackage{subcaption}
\usepackage{setspace}
\usepackage{enumitem}
\usepackage{multirow}
\usepackage{longtable}
\usepackage{tabularx}
\usepackage{float}
\usepackage{sectsty}
\usepackage{orcidlink}

% Pin footnotes to the bottom of every page, even on short pages
% (default LaTeX would float them up against the body text).
\usepackage[bottom]{footmisc}

\AtBeginEnvironment{tabular}{\onehalfspacing}

% Custom \maketitle for the Internet Appendix that matches the
% source draft layout: "Internet Appendix for" / title / authors /
% SUPPLEMENTARY RESULTS.
%
% IMPORTANT: the title comes from the YAML title: field in
% internet-appendix.qmd (via \@title) and stays in sync with the
% main paper automatically, but the AUTHOR NAMES BELOW ARE
% HARDCODED. If you add, remove, or rename authors in the main
% paper's index.qmd YAML, edit the \large {...} line below to
% match. (Quarto's structured author block isn't propagated into
% this partial because internet-appendix.qmd carries no by-author
% list of its own.)
\makeatletter
\renewcommand{\maketitle}{%
  \vspace*{0.25cm}%
  \begin{center}%
    {\LARGE Internet Appendix for}\\[0.50cm]%
    {\LARGE \@title}\\[0.75cm]%
    {\large Author One\textsuperscript{\normalsize\orcidlink{0000-0000-0000-0001}} \hspace{1cm} Author Two\textsuperscript{\normalsize\orcidlink{0000-0000-0000-0002}} \hspace{1cm} Author Three\textsuperscript{\normalsize\orcidlink{0000-0000-0000-0003}}}\\[1cm]%
    {\normalsize\textsc{Supplementary Results}}\\[1cm]%
  \end{center}%
  \normalsize%
}
\makeatother

% Italic subsections.
\subsectionfont{\large\itshape}
\subsubsectionfont{\large\itshape}
\makeatletter
\renewcommand{\subsection}{\@startsection{subsection}{2}{\z@}%
  {-3.25ex \@plus -1ex \@minus -.2ex}{1.5ex \@plus .2ex}%
  {\normalfont\large\itshape}}
\renewcommand{\subsubsection}{\@startsection{subsubsection}{3}{\z@}%
  {-3.25ex \@plus -1ex \@minus -.2ex}{1.5ex \@plus .2ex}%
  {\normalfont\large\itshape}}
\makeatother

% Period after section number.
\makeatletter
\def\@seccntformat#1{\@ifundefined{#1@cntformat}%
    {\csname the#1\endcsname\quad}%
    {\csname #1@cntformat\endcsname}%
}
\def\section@cntformat{\thesection.\quad}
\def\subsection@cntformat{\thesubsection \quad}
\makeatother

\tolerance=1
\emergencystretch=\maxdimen
\hyphenpenalty=10000
\hbadness=10000

% Symmetric display-equation spacing. See the comment in
% preamble.tex for the rationale; the four lengths are pinned to
% a single common value here and re-applied at \begin{document} so
% setspace / linestretch cannot drift them.
\setlength{\abovedisplayskip}{12pt}
\setlength{\belowdisplayskip}{12pt}
\setlength{\abovedisplayshortskip}{12pt}
\setlength{\belowdisplayshortskip}{12pt}
\AtBeginDocument{%
  \setlength{\abovedisplayskip}{12pt}%
  \setlength{\belowdisplayskip}{12pt}%
  \setlength{\abovedisplayshortskip}{12pt}%
  \setlength{\belowdisplayshortskip}{12pt}%
}

% Appendix labels: sections render as "Appendix A", "Appendix B",
% with tables / figures / equations as A.1, A.2, B.1, etc.
\renewcommand{\thesection}{Appendix \Alph{section}}
\renewcommand{\thesubsection}{\Alph{section}.\arabic{subsection}}
\renewcommand{\thetable}{\Alph{section}.\arabic{table}}
\renewcommand{\thefigure}{\Alph{section}.\arabic{figure}}
\renewcommand{\theequation}{\Alph{section}.\arabic{equation}}

\theoremstyle{plain}
\newtheorem{theorem}{Theorem}[section]
\newtheorem{proposition}[theorem]{Proposition}
\newtheorem{lemma}[theorem]{Lemma}
\theoremstyle{definition}
\newtheorem{definition}[theorem]{Definition}
\newtheorem{assumption}[theorem]{Assumption}
